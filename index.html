  // Skill Search Logic--------
const searchInput = document.getElementById("search-skills");
const tags = document.querySelectorAll("#skill-container .tag");
const noResultsEl = document.getElementById("no-skill");
const wrapper = document.getElementById("skills-wrapper");
// Add onkeyup function on search input
searchInput.onkeyup = (e) => {
  // Initialize RegExp with dynamic variable and case insensitive
  const regex = new RegExp(`${searchInput.value}`, "i");
  // Test each elementNode for regex match
  tags.forEach((tag) => tag.innerText.match(regex) ? tag.classList.remove("is-hidden") : tag.classList.add("is-hidden"));
  // If no matches return, show conditional element
  document.querySelectorAll("#skill-container .tag.is-hidden").length === document.querySelectorAll("#skill-container .tag").length ?
    (noResultsEl.classList.remove("is-hidden"), wrapper.classList.add("is-hidden"))
    : (noResultsEl.classList.add("is-hidden"), wrapper.classList.remove("is-hidden"));
};
